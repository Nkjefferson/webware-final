CREATE TABLE student(id SERIAL PRIMARY KEY, name text, profile text, img_url text);
CREATE TABLE professor(id SERIAL PRIMARY KEY, name text, profile text, img_url text);
CREATE TABLE course(id SERIAL PRIMARY KEY, name text, professor_id integer REFERENCES professor(id));
CREATE TABLE registration(student_id integer REFERENCES student(id), course_id integer REFERENCES course(id));
